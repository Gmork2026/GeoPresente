function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    
    // 1. Guardar siempre el registro en GeoPresente
    var sheetGeo = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("GeoPresente");
    sheetGeo.appendRow([
      new Date(),
      data.legajo,
      data.tipo_marcacion,
      data.id_servicio,
      data.descripcion_servicio,
      data.lat_celular,
      data.lon_celular,
      data.distancia_metros,
      data.es_valida,
      data.observaciones,
      data.selfie_url
    ]);

    // 2. Si es un usuario nuevo, guardarlo también en Vigiladores
    if (data.is_nuevo_usuario === true) {
      var sheetVig = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Vigiladores");
      sheetVig.appendRow([
        data.legajo,
        data.nombre_nuevo,
        data.dni_nuevo,
        data.selfie_url, // Por ahora dice PENDIENTE_N8N, luego lo actualizaremos
        "Sí"             // Habilitado por defecto
      ]);
    }

    return ContentService.createTextOutput(JSON.stringify({"status": "success"}))
                         .setMimeType(ContentService.MimeType.JSON);
                         
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "mensaje": error.toString()}))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  var action = e.parameter.action;
  var sheetName = action === "servicios" ? "Servicios" : (action === "vigiladores" ? "Vigiladores" : null);
  
  if (!sheetName) {
     return ContentService.createTextOutput(JSON.stringify({"error": "Acción inválida"}))
                          .setMimeType(ContentService.MimeType.JSON);
  }

  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var result = [];
  
  for (var i = 1; i < data.length; i++) {
    var obj = {};
    for (var j = 0; j < headers.length; j++) {
      obj[headers[j]] = data[i][j];
    }
    result.push(obj);
  }

  return ContentService.createTextOutput(JSON.stringify({"status": "success", "data": result}))
                       .setMimeType(ContentService.MimeType.JSON);
}
